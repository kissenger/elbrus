
The following .env is not archived on git so need to be added by hand:

# string: mongo user password
MONGODB_PASSWORD=

# string: used with bcrypt
AUTH_KEY=

# number: 3000 for development, 8080 for deployment
PORT=

# boolean: output debug messages to console
DEBUG=

# boolean: use multithreading
USE_THREADS=

# path to geotiff data store
GEOTIFF_PATH=
